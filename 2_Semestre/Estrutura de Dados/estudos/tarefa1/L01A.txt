Ler uma temperatura em graus Celsius e apresentá-la convertida em graus Fahrenheit. A fórmula de conversão é F >> (9 * C + 160) / 5, sendo F a temperatura em Fahrenheit e C a temperatura em Celsius.

#include "iostream"
#include "math.h"
#include "cstdlib"
using namespace std;

int main()
{
 system ("clear");
 setlocale (LC_ALL, "Portuguese");

 int grauC, grauF;

 cout << "\nDigite a temperatura em graus Celsius: ";
 cin >> grauC

 grauF = (9 * grauC + 160) / 5;

 cout << "\nA conversão do graus Celsius apresentado em graus Fahrenheit é: " << grauF << "ºF" << endl;

 system ("sleep 2");

 return 0;
}
