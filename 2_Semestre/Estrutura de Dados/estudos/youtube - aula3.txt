#include "iostream"
#include "cstdlib"
#include "math.h"
using namespace std;

double calcularMedia(double n1, double n2){
 double media;
 media = (n1 + n2)/2
 return media;
}

int main(){
 double nota1 = 10, nota2 = 4, valor_media;
 valor_media = calcularMedia(nota1,nota2);
 cout << valor_media;
 return 0;
}
